# rexray Cinder plugin.

```
swarm plugin install rexray/cinder:latest \
  CINDER_AUTHURL=https://identity.openstack.cloudvps.com/v3 \
  CINDER_USERNAME= \
  CINDER_PASSWORD= \
  CINDER_TENANTID= \
  CINDER_TENANTNAME="" \
  CINDER_DOMAINNAME=Default \
  CINDER_REGIONNAME=AMS \
  REXRAY_PREEMPT=true \
  REXRAY_LOGLEVEL=debug

```
